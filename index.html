function addPointToList(name, value, date) { const li = document.createElement("li"); // サービス名の色付け const nameSpan = document.createElement("span"); if(name.includes("楽天")) nameSpan.style.color = "#1E90FF"; else if(name.includes("Tポイント")) nameSpan.style.color = "#FF8C00"; else nameSpan.style.color = "#32CD32"; nameSpan.style.fontWeight = "bold"; nameSpan.textContent = name; const dateSpan = document.createElement("span"); dateSpan.style.color = "#555"; dateSpan.textContent = ` - ${date}: `; const valueSpan = document.createElement("span"); valueSpan.textContent = `${value}ポイント`; li.appendChild(nameSpan); li.appendChild(dateSpan); li.appendChild(valueSpan); // 削除ボタン const delBtn = document.createElement("button"); delBtn.textContent = "削除"; delBtn.style.background = "#f44336"; delBtn.addEventListener("click", () => { pointList.removeChild(li); savedPoints = savedPoints.filter(item => !(item.name === name && item.value === value && item.date === date)); localStorage.setItem("points", JSON.stringify(savedPoints)); updateTotals(); }); li.appendChild(delBtn); pointList.appendChild(li); }
