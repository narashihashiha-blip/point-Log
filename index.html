function addPointToList(name, value, date, color) { const li = document.createElement("li"); const nameSpan = document.createElement("span"); nameSpan.style.fontWeight = "bold"; nameSpan.style.color = color; // ←ここで色を反映 nameSpan.textContent = name; const dateSpan = document.createElement("span"); dateSpan.style.color = "#555"; dateSpan.textContent = ` - ${date}: `; const valueSpan = document.createElement("span"); valueSpan.textContent = `${value}ポイント`; li.appendChild(nameSpan); li.appendChild(dateSpan); li.appendChild(valueSpan); // 削除ボタン const delBtn = document.createElement("button"); delBtn.textContent = "削除"; delBtn.style.background = "#f44336"; delBtn.style.color = "#fff"; delBtn.style.marginLeft = "10px"; delBtn.addEventListener("click", () => { pointList.removeChild(li); savedPoints = savedPoints.filter(item => !(item.name === name && item.value === value && item.date === date)); localStorage.setItem("points", JSON.stringify(savedPoints)); updateTotals(); }); li.appendChild(delBtn); pointList.appendChild(li); }
