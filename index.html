function addPointToList(name, value, date) { const li = document.createElement("li"); // サービス名ごとに色分け let color = "#000"; // デフォルト黒 if(name.includes("楽天")) color = "#1E90FF"; // 青 else if(name.includes("Tポイント")) color = "#FF8C00"; // オレンジ else color = "#32CD32"; // 緑 li.innerHTML = `${name} - ${date}: ${value}ポイント`; const delBtn = document.createElement("button"); delBtn.textContent = "削除"; delBtn.style.background = "#f44336"; // 赤 delBtn.addEventListener("click", () => { pointList.removeChild(li); savedPoints = savedPoints.filter(item => !(item.name === name && item.value === value && item.date === date)); localStorage.setItem("points", JSON.stringify(savedPoints)); updateTotals(); }); li.appendChild(delBtn); pointList.appendChild(li); }
